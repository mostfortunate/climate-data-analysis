# Plot for 19th Century
set title "Low, High, and Average Temperatures (19th Century)"
set boxwidth 0.2
set style fill solid

set xrange [1850:2015]

plot "LandMinTemperatures.txt" using 1:2 with boxes lc rgb "blue" title "Low", \
     "LandMaxTemperatures.txt" using 1:2 with boxes lc rgb "green" title "High", \
     "YearlyLandAvgTemperatures.txt" using 1:2 with boxes lc rgb "red" title "Average"
